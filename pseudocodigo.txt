2. Pseudocódigo

 Algoritmo "Truque_da_Tabuada_do_9"

 Var
   n, dezena, unidade, resultado : Inteiro

 Inicio
   // Entrada de dados
   Escreva("Digite um número de 1 a 10: ")
   Leia(n)

   // Processamento
   dezena <- n - 1
   unidade <- 10 - n
   resultado <- (dezena * 10) + unidade

   // Saída de dados
   Escreva("O resultado de 9 x ", n, " é igual a: ", resultado)
 Fim Algoritmo
